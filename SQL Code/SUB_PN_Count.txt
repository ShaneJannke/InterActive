ALTER PROCEDURE [dbo].[SUB_PN_Count] 
	-- DECLARE
	@ID varchar(10)--Inputted WO with or without leading zeros
	,@iwo nvarchar(10)
AS
BEGIN
SET NOCOUNT ON;

SELECT COUNT (DISTINCT PN) AS PN_COUNT 
		,COUNT (DISTINCT WONO) AS WO_COUNT
FROM Interactive.dbo.sub_request
WHERE
@ID = REQUEST_ID 

END
