CREATE PROCEDURE [Update_TM_Feedback]
AS
BEGIN
SET NOCOUNT ON;

UPDATE TM_request
SET
PROD_FEEDBACK = CASE WHEN ipar.FEEDBACK like '' THEN 'N' ELSE 'Y' END 
FROM TM_request tm
INNER JOIN
MANEX.dbo.INVENTOR ipar
ON
tm.PCBA = ipar.PART_NO

END
